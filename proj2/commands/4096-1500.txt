#1. Sequential Read and Write Test
sudo fio --name=seq_readwrite \
         --filename=/sdb \
         --direct=1 \
         --rw=readwrite \
         --bs=4k \
         --numjobs=1 \
         --iodepth=1 \
         --time_based \
         --runtime=900 \
         --ramp_time=150 \
         --size=1G \
         --ioengine=libaio \
         --group_reporting \
         --stonewall \
         --output-format=csv \
         --write_hist_log=results/4096/1500/sequential/ \
         --write_lat_log=results/4096/1500/sequential/ \
         --write_bw_log=results/4096/1500/sequential/ \
         --write_iops_log=results/4096/1500/sequential/ \
         --output results/4096/1500/sequential/seq_read_write.log
